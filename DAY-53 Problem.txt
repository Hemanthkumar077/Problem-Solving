`Problem: Write a function to perform basic string compression 
using the counts of repeated characters. 
Testcase 1: 
Input: "aabcccccaaa" 
Output: "a2b1c5a3" 


def compress_string(s):
    if not s:
        return ""
    compressed = ""
    count = 1
    for i in range(1, len(s)):
        if s[i] == s[i - 1]:
            count += 1
        else:
            compressed += s[i - 1] + str(count)
            count = 1
    compressed += s[-1] + str(count)
    
    return compressed
print(compress_string("aabcccccaaa"))
